2Model User

Buat model User dengan fillable attributes:

namespace App\Models;

use Illuminate\Foundation\Auth\User as Authenticatable;

class User extends Authenticatable
{
    protected $fillable = [
        'name',
        'email',
        'password',
        'balance',
    ];

    protected $hidden = [
        'password',
        'remember_token',
    ];
}

3Controller Transaksi

Implementasi method topup di TransaksiController:

public function topup(Request $request)
{
    $request->validate([
        'user_id' => 'required|exists:users,id',
        'amount' => 'required|numeric|min:1'
    ]);

    $targetUser = User::findOrFail($request->user_id);

    $transaksi = Transaksi::create([
        'sender_id' => null,
        'receiver_id' => $targetUser->id,
        'tipe_transaksi' => 'topup',
        'confirmed' => true,
        'amount' => $request->amount
    ]);

    $targetUser->update([
        'balance' => $targetUser->balance + $request->amount
    ]);
    
    return redirect()->back()->with('success', 'Top up success');
}
Catatan: Pastikan semua dependencies sudah terinstall dan database sudah terkonfigurasi dengan benar.
